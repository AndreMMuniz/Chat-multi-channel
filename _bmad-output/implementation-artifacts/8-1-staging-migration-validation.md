## Known Issues Found in Audit

### Issue 1 — CONFIRMED BUG: `drop_constraint(None, ...)` in `bf57d943ecd0` downgrade

**File:** `backend/alembic/versions/bf57d943ecd0_add_rbac_and_missing_columns.py`

**Current code (lines 89, 96):**
```python
op.drop_constraint(None, 'messages', type_='foreignkey')   # line 89
op.drop_constraint(None, 'conversations', type_='foreignkey')  # line 96
```

**Problem:** Alembic autogenerate creates `None` constraint names when the original FK had no explicit name. PostgreSQL requires the real constraint name. Running `alembic downgrade` will fail with: `ProgrammingError: constraint None of relation "messages" does not exist`.

**Fix:** Replace `None` with the actual PostgreSQL constraint names. FK names in PostgreSQL autogenerated by Alembic/SA follow the pattern `<table>_<column>_fkey`:
- `messages_owner_id_fkey` — for `messages.owner_id → users.id`
- `conversations_assigned_user_id_fkey` — for `conversations.assigned_user_id → users.id`

### Issue 2 — KNOWN/ACCEPTABLE: `g2b3c4d5e6f7` PL/pgSQL enum creation

**File:** `backend/alembic/versions/g2b3c4d5e6f7_add_delivery_status_to_messages.py`

**Status:** Already fixed inline. The `DO $$ BEGIN ... IF NOT EXISTS ... END $$` block is valid PostgreSQL. The retro noted this had an issue; current code shows it was corrected. **No action needed.**

### Issue 3 — KNOWN/ACCEPTABLE: PostgreSQL partial indexes

**Files:** `g2b3c4d5e6f7` and `h3c4d5e6f7a8`

Partial indexes use `postgresql_where=sa.text(...)`. These are PostgreSQL-specific and will be ignored or fail on SQLite. **No action needed** since we target Supabase (PostgreSQL). The validation script should use PostgreSQL, not SQLite.

### Issue 4 — KNOWN/ACCEPTABLE: `bf57d943ecd0` downgrade recreates LangGraph tables

The downgrade recreates `checkpoint_blobs`, `checkpoints`, `checkpoint_migrations`, `checkpoint_writes`. These are LangGraph tables that were in the original DB when autogenerate ran but are not in the current app's models. The downgrade would restore them (harmless). **No action needed.**

### Task 2 — Create `backend/scripts/validate_migrations.py`

This script:
1. Verifies the migration chain is a clean linear sequence (no branches, no gaps).
2. Optionally runs `alembic upgrade head` against a target DATABASE_URL.
3. Runs a schema comparison against the SQLAlchemy models.
4. Prints a pass/fail report.

```python
#!/usr/bin/env python3
"""Validate Alembic migration chain integrity.

Usage:
    # Chain-only check (no DB required):
    python scripts/validate_migrations.py

    # Full upgrade + schema check against a real DB:
    DATABASE_URL=postgresql://... python scripts/validate_migrations.py --run-upgrade
"""

import os
import sys
import subprocess
import argparse

# Expected linear chain in order
EXPECTED_CHAIN = [
    ("de45c95bc364", None),
    ("bf57d943ecd0", "de45c95bc364"),
    ("c9f2e1d3a8b5", "bf57d943ecd0"),
    ("d7e3f9a1b2c4", "c9f2e1d3a8b5"),
    ("e8f4a2c6d9b1", "d7e3f9a1b2c4"),
    ("f1a2b3c4d5e6", "e8f4a2c6d9b1"),
    ("g2b3c4d5e6f7", "f1a2b3c4d5e6"),
    ("h3c4d5e6f7a8", "g2b3c4d5e6f7"),
    ("i4d5e6f7a8b9", "h3c4d5e6f7a8"),
]


def check_chain() -> bool:
    """Parse actual migration files and verify down_revision chain matches expected."""
    import importlib.util
    from pathlib import Path

    versions_dir = Path(__file__).parent.parent / "alembic" / "versions"
    actual: dict[str, str | None] = {}

    for path in versions_dir.glob("*.py"):
        if path.name.startswith("__"):
            continue
        spec = importlib.util.spec_from_file_location(path.stem, path)
        mod = importlib.util.module_from_spec(spec)
        try:
            spec.loader.exec_module(mod)
        except Exception as exc:
            print(f"  ERROR loading {path.name}: {exc}")
            return False
        rev = getattr(mod, "revision", None)
        down = getattr(mod, "down_revision", None)
        if rev:
            actual[rev] = down

    errors = []
    for rev, expected_down in EXPECTED_CHAIN:
        if rev not in actual:
            errors.append(f"  MISSING revision: {rev}")
        elif actual[rev] != expected_down:
            errors.append(
                f"  CHAIN ERROR {rev}: expected down_revision={expected_down!r}, "
                f"got {actual[rev]!r}"
            )

    extra = set(actual.keys()) - {r for r, _ in EXPECTED_CHAIN}
    if extra:
        errors.append(f"  UNEXPECTED revisions (not in expected chain): {extra}")

    if errors:
        for e in errors:
            print(e)
        return False
    return True


def run_upgrade() -> bool:
    """Run alembic upgrade head and return True on success."""
    result = subprocess.run(
        ["alembic", "upgrade", "head"],
        cwd=str(Path(__file__).parent.parent),
        capture_output=True,
        text=True,
    )
    print(result.stdout)
    if result.returncode != 0:
        print("UPGRADE FAILED:")
        print(result.stderr)
        return False
    return True


def run_downgrade_one() -> bool:
    """Run alembic downgrade -1 to test the last migration's downgrade."""
    result = subprocess.run(
        ["alembic", "downgrade", "-1"],
        cwd=str(Path(__file__).parent.parent),
        capture_output=True,
        text=True,
    )
    print(result.stdout)
    if result.returncode != 0:
        print("DOWNGRADE FAILED:")
        print(result.stderr)
        return False
    return True


def main():
    parser = argparse.ArgumentParser(description="Validate Alembic migration chain")
    parser.add_argument("--run-upgrade", action="store_true",
                        help="Run alembic upgrade head + schema check (requires DATABASE_URL)")
    parser.add_argument("--test-downgrade", action="store_true",
                        help="Also test downgrade -1 after upgrade (requires DATABASE_URL)")
    args = parser.parse_args()

    print("=" * 60)
    print("Migration Chain Integrity Check")
    print("=" * 60)

    print(f"\nChecking chain of {len(EXPECTED_CHAIN)} migrations...")
    ok = check_chain()
    if ok:
        print(f"  ✅ Chain is valid ({len(EXPECTED_CHAIN)} migrations, linear, no gaps)")
    else:
        print("  ❌ Chain has errors (see above)")
        sys.exit(1)

    if args.run_upgrade:
        db_url = os.getenv("DATABASE_URL", "")
        if not db_url:
            print("\n❌ DATABASE_URL not set — cannot run upgrade")
            sys.exit(1)
        print(f"\nRunning: alembic upgrade head")
        print(f"Target: {db_url[:40]}...")
        if not run_upgrade():
            sys.exit(1)
        print("  ✅ upgrade head completed")

        if args.test_downgrade:
            print("\nRunning: alembic downgrade -1 (rollback test)")
            if not run_downgrade_one():
                sys.exit(1)
            print("  ✅ downgrade -1 completed")
            print("\nRe-running: alembic upgrade head (restore after rollback test)")
            if not run_upgrade():
                sys.exit(1)
            print("  ✅ restore completed")

    print("\n" + "=" * 60)
    print("✅ All checks passed")
    print("=" * 60)


if __name__ == "__main__":
    main()
```
